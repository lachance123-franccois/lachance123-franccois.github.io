Exemple de Code 

% Filtre Adaptatif Wiener - Exemple MATLAB
clear; clc;

% Charger l'image
img = imread('image.jpg');
img_gray = rgb2gray(img);

% Ajouter du bruit
img_noisy = imnoise(img_gray, 'gaussian', 0, 0.01);

% Appliquer filtre Wiener
windowSize = [5, 5];
img_filtered = wiener2(img_noisy, windowSize);

% Afficher résultats
figure;
subplot(1,3,1), imshow(img_gray), title('Original');
subplot(1,3,2), imshow(img_noisy), title('Avec Bruit');
subplot(1,3,3), imshow(img_filtered), title('Filtré (Wiener)');

% Calculer PSNR (Peak Signal-to-Noise Ratio)
psnr_value = psnr(img_gray, img_filtered);
fprintf('PSNR: %.2f dB\n', psnr_value);
